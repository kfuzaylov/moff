Create for mobile, extend to desktop.

Create light, fast and best optimized websites for mobile devices first

Easily extend your website from mobile to tablet and desktop screens

Rich API, Modularity system, Data Events and AMD make your development easy

Moff is a JavaScript framework designed to offer best practices for mobile first development. It has rich API to work with browser, files and modularity system. Also it support Asynchronous Module Definition (AMD) which help to control loading of your modules. Data events system allows to load content with the least efforts. And it has pluggable system which allows to create plugins and extend the framework.

Moff.amd.register({
	id: 'thumbsModule',
	depend: {
		js: ['bower_components/jquery/dist/jquery.min.js', 'bower_components/slick-carousel/slick/slick.min.js'],
		css:['bower_components/slick-carousel/slick/slick.css']
	},
	file: {
		js: ['javascripts/thumbs.js']
	}
});
// thumbs.js

// jQuery and Slick carousel are already loaded
$('.image-thumbs').slick({
	infinite: true,
	slidesToShow: 3,
	slidesToScroll: 3
});

$('.image-thumbs img').on('click', function() {
	var number = $(this).attr('src').match(/thumb(\d)\.png/)[1];
	$('.main-image img').attr('src', 'images/preview' + number + '.png');
});
